Hash Table

A key-value store using a hash function to compute bucket indices. Collisions are resolved via chaining (linked lists at each bucket). Average O(1) insert, search & delete.

Hash Table Code
class HashTable { constructor(size = 11) { this.size = size; this.table = Array.from( { length: size }, () => [] ); } hash(key) { let h = 0; for (let c of key) h = (h * 31 + c.charCodeAt(0)) % this.size; return h; } insert(key, val) { const idx = this.hash(key); const bucket = this.table[idx]; const ex = bucket.findIndex(([k]) => k === key); if (ex !== -1) bucket[ex][1] = val; else bucket.push([key, val]); } search(key) { const idx = this.hash(key); return this.table[idx].find(([k]) => k === key)?.[1]; } delete(key) { const idx = this.hash(key); this.table[idx] = this.table[idx].filter(([k]) => k !== key); } }
