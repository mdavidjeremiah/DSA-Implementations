class Graph { constructor() { this.adjList = new Map(); } // Add vertex — O(1) addVertex(v) { if (!this.adjList.has(v)) this.adjList.set(v, []); } // Add undirected edge — O(1) addEdge(u, v) { this.addVertex(u); this.addVertex(v); this.adjList.get(u).push(v); this.adjList.get(v).push(u); } // BFS — O(V + E) BFS(start) { const vis = new Set(), q = [start], o = []; vis.add(start); while (q.length) { const n = q.shift(); o.push(n); for (const nb of this.adjList.get(n) ?? []) if (!vis.has(nb)) { vis.add(nb); q.push(nb); } } return o; } // DFS — O(V + E) DFS(start) { const vis = new Set(), o = []; const dfs = v => { vis.add(v); o.push(v); for (const nb of this.adjList.get(v) ?? []) if (!vis.has(nb)) dfs(nb); }; dfs(start); return o; } }
