// Linear Search Code
function linearSearch(arr, target) { for (let i = 0; i < arr.length; i++) { // Check each element one by one if (arr[i] === target) { return i; // Found at index i } } return -1; // Not found } /* Works on UNSORTED arrays. Scans left → right until found or end is reached. */
// Binary Search Code
function binarySearch(arr, target) { let low = 0, high = arr.length - 1; while (low <= high) { const mid = Math.floor((low+high) / 2); if (arr[mid] === target) return mid; else if (arr[mid] < target) low = mid + 1; else high = mid - 1; } return -1; // Not found } /* Requires SORTED array. Halves search space per step. */
