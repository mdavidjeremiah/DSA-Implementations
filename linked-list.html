Linked List ADT

A linear collection of nodes where each node holds a value and a pointer to the next node. Dynamic size with O(1) head insertion. No index-based random access — traversal is sequential.

ADT Implementation (JS)
class Node { constructor(val) { this.val = val; this.next = null; } } class LinkedList { constructor() { this.head = null; this.size = 0; } // Insert at head — O(1) insertHead(val) { const node = new Node(val); node.next = this.head; this.head = node; this.size++; } // Insert at tail — O(n) insertTail(val) { const node = new Node(val); if (!this.head) { this.head = node; } else { let cur = this.head; while (cur.next) cur = cur.next; cur.next = node; } this.size++; } // Delete by value — O(n) delete(val) { if (!this.head) return false; if (this.head.val === val) { this.head = this.head.next; this.size--; return true; } let cur = this.head; while (cur.next) { if (cur.next.val === val) { cur.next = cur.next.next; this.size--; return true; } cur = cur.next; } return false; } // Search by value — O(n) search(val) { let cur = this.head, idx = 0; while (cur) { if (cur.val === val) return idx; cur = cur.next; idx++; } return -1; } }
