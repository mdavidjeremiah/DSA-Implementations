Interpolation Search

An improved binary search for uniformly distributed sorted arrays. Instead of always picking the midpoint, it estimates the target's position proportionally — like how you'd look up a word in a dictionary. Average O(log log n), worst O(n).

Interpolation Search Code
function interpolationSearch(arr, target) { let low = 0, high = arr.length - 1; while (low <= high && target >= arr[low] && target <= arr[high]) { if (arr[low] === arr[high]) { return arr[low] === target ? low : -1; } const probe = low + Math.floor( ((target - arr[low]) / (arr[high] - arr[low])) * (high - low) ); if (arr[probe] === target) return probe; else if (arr[probe] < target) low = probe + 1; else high = probe - 1; } return -1; }
