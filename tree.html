Binary Search Tree ADT

A hierarchical structure where each node has at most two children. The BST invariant: left subtree values < node < right subtree values. Enables O(log n) search, insert & delete on balanced trees.

ADT Implementation
class BSTNode { constructor(val) { this.val = val; this.left = null; this.right = null; } } class BST { constructor() { this.root = null; } insert(val) { this.root = this._insert(this.root, val); } _insert(node, val) { if (!node) return new BSTNode(val); if (val < node.val) node.left = this._insert(node.left, val); else if (val > node.val) node.right = this._insert(node.right, val); return node; } search(val) { let cur = this.root; while (cur) { if (val === cur.val) return true; cur = val < cur.val ? cur.left : cur.right; } return false; } inorder() { const res = []; const dfs = n => { if (!n) return; dfs(n.left); res.push(n.val); dfs(n.right); }; dfs(this.root); return res; } }
